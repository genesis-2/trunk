run --name=<job-name> [--async-name=<rns-path>] 
	{<bes-container> | <scheduler>} [run-options] <executable> [program-args]

	OR
	
run [--name=<job-name>] [--async-name=<rns-path>] --jsdl=<jsdl-file> 
	{<bes-container> | <scheduler>}

	OR
	
run --check-status <job-path>

WHERE run-options are:
	--stdout=<out-filename>
	--stderr=<err-filename>
	--stdin=<in-filename>
	--stage-in=<filename>/<stage-in-uri>
	--stage-out=<filename>/<stage-out-uri>
	--D<env-var>=<env-value>
